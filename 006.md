#include <string>
#include <iostream>

struct Nec {
	Nec(const std::string&)
	{
		std::cout << '1';
	}

	template <typename T>
	Nec(T&&)
	{
		std::cout << '2';
	}
};

int main()
{
	std::string s{ "a" };
	const Nec n1{ "a" };
	Nec n2(s);
	Nec n3(move(s));
	Nec n4(n1);
	Nec n5(n2);
}
