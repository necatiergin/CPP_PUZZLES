#include <vector>
#include <set>
#include <iostream>

int main()
{
	std::vector v1{ 3, 2 };
	decltype(v1) v2(3, 2);
	decltype(v2) v3((7, 3));
	std::vector v4{ v2.begin(), v2.end() };
	std::set<std::size_t> s{ 
		v1.size(), 
		v2.size(), 
		v3.size(), 
		v4.size()
	};

	std::cout << v1.size() 
		<< v2.size() 
		<< v3.size()
		<< v4.size() 
		<< s.size();
}
